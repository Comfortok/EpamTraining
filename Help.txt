package com.epamtraining.stream27;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.io.PrintWriter;

@WebServlet("/AutorizationServlet")
public class AutorizationServlet extends HttpServlet {
    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        doGet(request, response);
    }

    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
        response.setContentType("text/html");
        String parameterFirst = request.getParameter("username");
        String parameterSecond = request.getParameter("password");
        if (parameterFirst == null || parameterSecond == null){
        } else if (("".equals(parameterFirst)) || ("".equals(parameterSecond))){
        } else {
            System.out.println("Username? = " + parameterFirst + ", Password? = " + parameterSecond);
            PrintWriter out = response.getWriter();
            out.write("<html><body><div id='servletResponse' style='text-align: center;'>");
            if ((parameterFirst.equalsIgnoreCase("root")) && (parameterSecond.equals("pass"))){
                out.write("<h2>Servlet Application Login Sample</h2>");
                out.write("<p style='color: green; font-size: large;'> Congratulations! <span style='text-transform: capitalize;'>" + parameterSecond + "</span>, You are an authorised login!</p>");
            } else {
                out.write("<p style='color: red; font-size: larger;'>You are not authorized! Check your login or pass.</p>");
            }
            out.write("</div></body></html>");
            out.close();
        }
    }
}
